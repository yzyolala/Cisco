class Solution {
    public List<List<Integer>> combinationSum(int[] candidates, int target) {
        List<List<Integer>> res= new ArrayList<>();
        bt(res,candidates,target,0,0,new LinkedList<>());//注意这里path传入的是LinkedList,而不是ArrayList
        return res;
    }

    private void bt(List<List<Integer>> res,int[] candidates,int target,int index,int sum,List<Integer> path){
        if(sum==target){
            res.add(new LinkedList(path));//注意这里path传入的是LinkedList,而不是ArrayList
        }
        for(int i=index;i<candidates.length;i++){
            if(sum+candidates[i]<=target){
                path.add(candidates[i]);
                bt(res,candidates,target,i,sum+candidates[i],path);//注意这里是candidates[i]不是i+1
                path.remove(path.size()-1);
            }
        }
    }
}
